254 static void SpiReadAfterHeader(void)
255 {
256  long data_to_recv = 0;
257  unsigned char *evnt_buff, type;
258 
259  /* Determine what type of packet we have */
260  evnt_buff = spiInformation.pRxPacket;
261  STREAM_TO_UINT8((char *)(evnt_buff + SPI_HEADER_SIZE),
262  HCI_PACKET_TYPE_OFFSET, type);
263 
264  switch(type)
265  {
266  case HCI_TYPE_DATA:
267  {
268  /* We need to read the rest of data.. */
269  STREAM_TO_UINT16((char *)(evnt_buff + SPI_HEADER_SIZE),
270  HCI_DATA_LENGTH_OFFSET, data_to_recv);
271 
272  if (!((CC3000_HEADERS_SIZE_EVNT + data_to_recv) & 1))
273  {
274  data_to_recv++;
275  }
276 
277  if (data_to_recv)
278  {
279  SpiReadDataSynchronous(evnt_buff + CC3000_SPI_MIN_READ_B,
280  data_to_recv);
281  }
282  break;
283  }
284  case HCI_TYPE_EVNT:
285  {
286  /* Calculate the rest length of the data*/
287  STREAM_TO_UINT8((char *)(evnt_buff + SPI_HEADER_SIZE),
288  HCI_EVENT_LENGTH_OFFSET, data_to_recv);
289  data_to_recv -= 1;
290 
291  /* Add padding byte if needed */
292  if ((CC3000_HEADERS_SIZE_EVNT + data_to_recv) & 1)
293  {
294  data_to_recv++;
295  }
296 
297  if (data_to_recv)
298  {
299  SpiReadDataSynchronous(evnt_buff + CC3000_SPI_MIN_READ_B,
300  data_to_recv);
301  }
302 
303  break;
304  }
305  }
306 }

328 static msg_t irqSignalHandlerThread(void *arg)
329 {
330  (void)arg;
331 
332 #if CH_USE_REGISTRY == TRUE
333  chRegSetThreadName(__FUNCTION__);
334 #endif
335 
336  while (1)
337  {
338  /* Wait here until the EXT interrupt signals that the IRQ line went
339  * low. */
340  CHIBIOS_CC3000_DBG_PRINT("IRQ waiting on semaphore.", NULL);
341  chBSemWait(&irqSem);
342 
343  CHIBIOS_CC3000_DBG_PRINT("IRQ waiting on pause.", NULL);
344 
345  while (spiPaused == true)
346  {
347  chThdSleep(10);
348  }
349 
350  CHIBIOS_CC3000_DBG_PRINT("IRQ Running.", NULL);
351 
352  if (spiInformation.spiState == SPI_STATE_POWERUP)
353  {
354  /* This means IRQ line was low call a callback of HCI Layer to inform on event */
355  spiInformation.spiState = SPI_STATE_INITIALIZED;
356  }
357 
358  else if (spiInformation.spiState == SPI_STATE_IDLE)
359  {
360 
361  spiInformation.spiState = SPI_STATE_READ;
362 
363  /* IRQ line goes down - start reception */
364  selectCC3000();
365 
366  SpiReadHeader();
367 
368  SpiReadAfterHeader();
369 
374  chThdSleep(MS2ST(100));
375 
376  SpiTriggerRxProcessing();
377  }
378 
379  else if (spiInformation.spiState == SPI_STATE_WRITE)
380  {
381  SpiWriteDataSynchronous(spiInformation.pTxPacket,
382  spiInformation.txPacketLength);
383 
384  spiInformation.spiState = SPI_STATE_IDLE;
385 
386  unselectCC3000();
387  }
388  }
389 
390  return 0;
391 }

454 void SpiWrite(unsigned char *pUserBuffer, unsigned short usLength)
455 {
456  /* If usLength is even, we need to add padding byte */
457  if (!(usLength & 0x01))
458  {
459  usLength++;
460  }
461 
462  /* @todo TI issue: Why can't altering pUserBuffer and usLength be done
463  * in the host driver? */
464 
465  /* Fill in SPI header */
466  pUserBuffer[CC3000_SPI_INDEX_OP] = CC3000_SPI_OP_WRITE;
467  pUserBuffer[CC3000_SPI_INDEX_LEN_MSB] = ((usLength) & 0xFF00) >> 8;
468  pUserBuffer[CC3000_SPI_INDEX_LEN_LSB] = ((usLength) & 0x00FF);
469  pUserBuffer[CC3000_SPI_INDEX_BUSY_1] = CC3000_SPI_BUSY;
470  pUserBuffer[CC3000_SPI_INDEX_BUSY_2] = CC3000_SPI_BUSY;
471 
472  usLength += SPI_HEADER_SIZE;
473 
474  if (wlan_tx_buffer[CC3000_SPI_TX_MAGIC_INDEX] != CC3000_SPI_MAGIC_NUMBER)
475  {
476  CHIBIOS_CC3000_DBG_PRINT("Buffer overflow detected.", NULL);
477  while(1);
478  }
479 
480  if (spiInformation.spiState == SPI_STATE_POWERUP)
481  {
482  while (spiInformation.spiState != SPI_STATE_INITIALIZED);
483  }
484 
485  if (spiInformation.spiState == SPI_STATE_INITIALIZED)
486  {
487  SpiFirstWrite(pUserBuffer, usLength);
488  }
489  else
490  {
491  /* We need to prevent here race that can occur in case 2 back to back
492  * packets are sent to the device, so the state will move to IDLE and
493  * once again to not IDLE due to IRQ */
494  tSLInformation.WlanInterruptDisable();
495 
496  while (spiInformation.spiState != SPI_STATE_IDLE);
497 
498  spiInformation.spiState = SPI_STATE_WRITE;
499  spiInformation.pTxPacket = pUserBuffer;
500  spiInformation.txPacketLength = usLength;
501 
502  /*Assert the CS line and wait till SSI IRQ line is active and then
503  * initialize write operation*/
504  selectCC3000();
505 
506  /*Re-enable IRQ */
507  tSLInformation.WlanInterruptEnable();
508  chThdYield();
509  }
510 
511  /* Due to the fact that we are currently implementing a blocking situation
512  here we will wait till end of transaction.*/
513  while (SPI_STATE_IDLE != spiInformation.spiState);
514 }
