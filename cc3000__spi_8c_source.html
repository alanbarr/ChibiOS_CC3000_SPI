349 static void SpiReadAfterHeader(void)
350 {
351  long data_to_recv = 0;
352  unsigned char *evnt_buff, type;
353 
354  /* Determine what type of packet we have */
355  evnt_buff = spiInformation.pRxPacket;
356  STREAM_TO_UINT8((char *)(evnt_buff + SPI_HEADER_SIZE),
357  HCI_PACKET_TYPE_OFFSET, type);
358 
359  switch(type)
360  {
361  case HCI_TYPE_DATA:
362  {
363  /* We need to read the rest of data.. */
364  STREAM_TO_UINT16((char *)(evnt_buff + SPI_HEADER_SIZE),
365  HCI_DATA_LENGTH_OFFSET, data_to_recv);
366 
367  if (!((CC3000_HEADERS_SIZE_EVNT + data_to_recv) & 1))
368  {
369  data_to_recv++;
370  }
371 
372  if (data_to_recv)
373  {
374  SpiReadDataSynchronous(evnt_buff + CC3000_SPI_MIN_READ_B,
375  data_to_recv);
376  }
377  break;
378  }
379  case HCI_TYPE_EVNT:
380  {
381  /* Calculate the rest length of the data*/
382  STREAM_TO_UINT8((char *)(evnt_buff + SPI_HEADER_SIZE),
383  HCI_EVENT_LENGTH_OFFSET, data_to_recv);
384  data_to_recv -= 1;
385 
386  /* Add padding byte if needed */
387  if ((CC3000_HEADERS_SIZE_EVNT + data_to_recv) & 1)
388  {
389  data_to_recv++;
390  }
391 
392  if (data_to_recv)
393  {
394  SpiReadDataSynchronous(evnt_buff + CC3000_SPI_MIN_READ_B,
395  data_to_recv);
396  }
397 
398  break;
399  }
400  }

424 {
425  (void)arg;
426 
427 #if CH_USE_REGISTRY == TRUE
428  chRegSetThreadName(__FUNCTION__);
429 #endif
430 
431  while (1)
432  {
433  /* Wait here until the EXT interrupt signals that the IRQ line went
434  * low. */
435  CHIBIOS_CC3000_DBG_PRINT("IRQ waiting on semaphore.", NULL);
436 
437  chSemWait(&irqSem);
438 
439  CHIBIOS_CC3000_DBG_PRINT("IRQ waiting on pause.", NULL);
440 
441  while (spiPaused == true)
442  {
443  chThdSleep(5);
444  }
445 
446  if (chThdShouldTerminate())
447  {
448  break;
449  }
450 
451  CHIBIOS_CC3000_DBG_PRINT("IRQ Running.", NULL);
452 
453  while (spiInformation.spiState != SPI_STATE_POWERUP &&
454  spiInformation.spiState != SPI_STATE_IDLE &&
455  spiInformation.spiState != SPI_STATE_WRITE_REQUESTED)
456  {
457  chThdSleep(5); /* XXX can this happen?? - yes.
458  Witnessed the while loop being hit once while the
459  state was initialised */
460  }
461 
462  if (spiInformation.spiState == SPI_STATE_POWERUP)
463  {
464  /* This means IRQ line was low call a callback of HCI Layer to inform on event */
465  setSpiState(SPI_STATE_INITIALIZED);
466  }
467 
468  else if (spiInformation.spiState == SPI_STATE_IDLE)
469  {
470  setSpiState(SPI_STATE_READ);
471 
472  /* IRQ line goes down - start reception */
473  selectCC3000();
474 
475  SpiReadHeader();
476 
477  SpiReadAfterHeader();
478 
479 #if 1
480 
484  chThdSleep(MS2ST(100));
485 #endif
486 
487  SpiTriggerRxProcessing();
488  }
489

491  {
492  setSpiState(SPI_STATE_WRITE_PERMITTED);
493 
494  }
495  }
496 
497  return 0;

560 void SpiWrite(unsigned char *pUserBuffer, unsigned short usLength)
561 {
562  /* If usLength is even, we need to add padding byte */
563  if (!(usLength & 0x01))
564  {
565  usLength++;
566  }
567 
568  /* @todo TI issue: Why can't altering pUserBuffer and usLength be done
569  * in the host driver? */
570 
571  /* Fill in SPI header */
572  pUserBuffer[CC3000_SPI_INDEX_OP] = CC3000_SPI_OP_WRITE;
573  pUserBuffer[CC3000_SPI_INDEX_LEN_MSB] = ((usLength) & 0xFF00) >> 8;
574  pUserBuffer[CC3000_SPI_INDEX_LEN_LSB] = ((usLength) & 0x00FF);
575  pUserBuffer[CC3000_SPI_INDEX_BUSY_1] = CC3000_SPI_BUSY;
576  pUserBuffer[CC3000_SPI_INDEX_BUSY_2] = CC3000_SPI_BUSY;
577 
578  usLength += SPI_HEADER_SIZE;
579 
580  if (wlan_tx_buffer[CC3000_SPI_TX_MAGIC_INDEX] != CC3000_SPI_MAGIC_NUMBER)
581  {
582  CHIBIOS_CC3000_DBG_PRINT("Buffer overflow detected.", NULL);
583  while(1);
584  }
585 
586  if (spiInformation.spiState == SPI_STATE_POWERUP)
587  {
588  while (spiInformation.spiState != SPI_STATE_INITIALIZED);
589  }
590 
591  if (spiInformation.spiState == SPI_STATE_INITIALIZED)
592  {
593  SpiFirstWrite(pUserBuffer, usLength);
594  }
595  else
596  {
597  /* We need to prevent here race that can occur in case 2 back to back
598  * packets are sent to the device, so the state will move to IDLE and
599  * once again to not IDLE due to IRQ */
600  tSLInformation.WlanInterruptDisable();
601 
602  while (spiInformation.spiState != SPI_STATE_IDLE); /* TODO sleep */
603 
604  setSpiState(SPI_STATE_WRITE_REQUESTED);
605  spiInformation.pTxPacket = pUserBuffer;
606  spiInformation.txPacketLength = usLength;
607 
608  /* Assert the CS line and wait till SSI IRQ line is active and then
609  * initialize write operation*/
610  selectCC3000();
611 
612  /*Re-enable IRQ */
613  tSLInformation.WlanInterruptEnable();
614  chThdYield();
615 
616  while (spiInformation.spiState != SPI_STATE_WRITE_PERMITTED); /* TODO sleep */
617 
618  SpiWriteDataSynchronous(spiInformation.pTxPacket,
619  spiInformation.txPacketLength);
620 
621  setSpiState(SPI_STATE_IDLE);
622 
623  unselectCC3000();
624  }
625 
626  /* Due to the fact that we are currently implementing a blocking situation
627  here we will wait till end of transaction.*/
628  while (SPI_STATE_IDLE != spiInformation.spiState);
629 }
